<script lang="ts" setup>
import { ref, onMounted } from 'vue'

interface Event {
  id: string
  summary: string
  description: string
  start: string
  end: string
}

const events = ref<Event[]>([])
const nextEvents = ref<Event[]>([])
const currentTime = ref<string>('')
const backgroundImage = ref<string>('https://via.placeholder.com/1920x1080')

// Fetch events and select the next two
const fetchEvents = async () => {
  try {
    const response = await fetch('http://localhost:3000/schedule')
    const allEvents = await response.json()
    events.value = allEvents
    nextEvents.value = allEvents.slice(0, 2) // Select only the next 2 events
  } catch (error) {
    console.error('Error fetching events:', error)
  }
}

// Update the current time every second
const updateTime = () => {
  currentTime.value = new Date().toLocaleTimeString([], {
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit',
  })
  setTimeout(updateTime, 1000)
}

// Format date as "15. februar"
const formatDate = (dateString: string): string => {
  const options: Intl.DateTimeFormatOptions = { day: 'numeric', month: 'long' }
  return new Intl.DateTimeFormat('no-NO', options).format(new Date(dateString))
}

// Format time as "20:00"
const formatTime = (dateString: string): string => {
  return new Date(dateString).toLocaleTimeString([], {
    hour: '2-digit',
    minute: '2-digit',
  })
}

// Lifecycle hook
onMounted(() => {
  fetchEvents()
  updateTime()
})
</script>

<template>
  <div class="min-h-screen bg-cover bg-center relative">
    <div class="backdrop-blur-md bg-black/30 min-h-screen flex flex-col justify-center px-8">
      <!-- Current Time -->
      <h1 class="text-white text-5xl font-bold pb-6">{{ currentTime }}</h1>

      <!-- Events -->
      <div class="space-y-4">
        <div v-for="event in nextEvents" :key="event.id" class="space-y-1">
          <p class="text-lg font-bold">
            {{ formatDate(event.start) }} | {{ formatTime(event.start) }}
          </p>
          <h2 class="text-xl font-semibold">{{ event.summary }}</h2>
          <p class="text-sm">{{ event.description }}</p>
          <p class="text-sm">
            Timeslot: {{ formatTime(event.start) }} - {{ formatTime(event.end) }}
          </p>
        </div>
      </div>

      <!-- Footer -->
      <div class="absolute bottom-4 left-8 text-white text-xs">
        <p>

          Forlag. All rights reserved.
        </p>
      </div>

      <!-- Top Right Icon -->
      <div class="absolute top-4 right-4">
        <img src="https://via.placeholder.com/50" alt="Logo" class="h-12 w-12" />
      </div>
    </div>
  </div>
</template>
